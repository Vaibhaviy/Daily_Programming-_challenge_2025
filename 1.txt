#include<iostream>
#include<vector>
using namespace std;

void sortNumbers(vector<int> &nums){
    int n=nums.size();
    int low=0,mid=0,high=n-1;
    while(mid<=high){
        if(nums[mid]==0){
            swap(nums[mid],nums[low]);
            low++; mid++;
        }else if(nums[mid]==1){
            mid++;
        }else{  
             swap(nums[mid],nums[high]);
                high--;
        }
    }
}

int main(){
    vector<int>nums={0,1,0,2,2,0,1,0,2,1,0,0,1,1,2,1};
    cout<<"Input array is:";
    for(int i=0;i<nums.size();i++){
        cout<<nums[i]<<" ";}
        cout<<endl;
    sortNumbers(nums);
     cout<<"Sorted array is:";
    for(int i=0;i<nums.size();i++){
        cout<<nums[i]<<" ";
    }

 return 0;
}
