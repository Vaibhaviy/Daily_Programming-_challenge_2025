
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

string longestPalindrome(string s) {
    int n = s.length();
    if (n <= 1) return s;

    string ans = s.substr(0, 1);

    for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
        string substr = s.substr(i, j - i + 1);
          string rev = substr;
         reverse(rev.begin(), rev.end());

            
        if (substr == rev && substr.length() > ans.length()) {
                ans = substr;
       }
    }
    }
    return ans;
}

int main() {
    string s = "abc";
    cout << longestPalindrome(s) << endl;
    
    
    
    return 0;
}
