#include <iostream>
#include <string>
#include <unordered_map>
using namespace std;


int atMostK(const string &s, int k) {
    int n = s.length();
  int left = 0;
unordered_map<char, int> freq;
int cnt = 0;

    for (int right = 0; right < n; right++) {
        freq[s[right]]++;
        while ((int)freq.size() > k) {
        freq[s[left]]--;
            if (freq[s[left]] == 0) {
                freq.erase(s[left]);
          }
            left++;
        }

        
        cnt += (right - left + 1);
  } return cnt;
}


int countSubstrExactlyK(const string &s, int k) {
    if (k == 0 || k > (int)s.size()) return 0; 
    return atMostK(s, k) - atMostK(s, k - 1);
}
