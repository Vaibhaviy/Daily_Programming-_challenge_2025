#include <iostream>
using namespace std;

int gcd(int a,int b){
if(b==0)  return a;

return gcd(b,a%b);
}

int lcm(int a,int b){
if(a==1) return b;
if(b==1) return a;
if(a==b) return a
    
    
int GCD=gcd(a,b);
return (a*b)/GCD;
}


int main(){
int m,n;
cout<<"Enter first number :";
cin>>m;
// cout<<endl;
cout<<"Enter second number :";
cin>>n;

cout<<"Least common multiple of given two numbers is :"<<lcm(m,n)<<endl;



return 0;
}
