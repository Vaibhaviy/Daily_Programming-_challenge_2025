#include <iostream>
#include <sstream>
#include<string>
#include<stack>
#include<cmath>
using namespace std;

int postfixEvaluation(string s){
stack<int>st;

istringstream iss(s);
    string token;

while (iss >> token) {
        if (isdigit(token[0]) || (token[0] == '-' && token.size() > 1)) {
            st.push(stoi(token));  
        }
else{

int opt2=st.top();
   st.pop();
int opt1=st.top();
   st.pop();

char op=token[0];
  switch(op){
case'+':
st.push(opt1+opt2);
break;
case'-':
st.push(opt1-opt2);
break;
case'*':
st.push(opt1*opt2);
break;
case'/':
st.push(opt1/opt2);
break;
case'^':
st.push(pow(opt1,opt2));
break;
}}} 
return st.top();
}


int main(){

string s="5 6 +";
cout<<"Answer of this postfix expression is :"<<postfixEvaluation(s)<<endl;
return 0;}
