#include<iostream>
#include<vector>

using namespace std;

int missingNumber(vector<int>&Arr){
    int n=Arr.size();
    if(n==1 && Arr[0]==1){ return 2;}
    if(Arr[n-1]!=n+1){return n+1;}
    
    
    for(int i=0;i<n;i++){
        if(Arr[i]!=i+1){
            return i+1;
        }
    }
    return -1;
}

int main(){
    vector<int>nums={1,2,3,5};
    int missing_number=missingNumber(nums);
    cout<<"Missing number in an array is:"<<missing_number<<endl;

    

 return 0;
}
