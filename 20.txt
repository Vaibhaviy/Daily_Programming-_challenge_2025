#include <iostream>
#include <stack>
using namespace std;

void insert(stack<int>& st, int x);

void sortStack(stack<int>& st) {
    if (st.empty()) return;
 int x = st.top();
    st.pop();
    sortStack(st);
    insert(st, x);
}

void insert(stack<int>& st, int x) {
    if (st.empty() || x >= st.top()) {
        st.push(x);
        return;
}   int temp = st.top();
    st.pop();
    insert(st, x);
    st.push(temp);
}

int main() {
    stack<int> st;
    st.push(3);
    st.push(2);
    st.push(1);
    st.push(4);

    sortStack(st);
     cout<<"Stack after sorting (top to bottom) :";
    while (!st.empty()) {
        cout << st.top() << " ";
        st.pop();
    }
    return 0;
}
