#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

int firstElementToRepeatKTimes(vector<int>& arr, int k) {
    unordered_map<int, int> freq;

    
    for (int num : arr) {
      freq[num]++;
    }
   for (int num : arr) {
        if (freq[num] == k) {
        return num;
        }
    }

    return -1;
}

int main() {
    vector<int> arr = {2, 3, 4, 2, 2, 5, 5};
    int k = 2;

int result = firstElementToRepeatKTimes(arr, k);
cout << "Output: " << result << endl;
    return 0;
}
