#include<iostream>
#include<vector>
using namespace std;



int duplicate(vector<int>& nums) {
        int slow=nums[0],fast=nums[0];
        do{
            slow=nums[slow];
            fast=nums[nums[fast]];
        } while(slow!=fast);
         slow=nums[0];

         while(slow!=fast){
             slow=nums[slow];
            fast=nums[fast];
         }
        return slow;
    }

int main(){
vector<int>Arr={1,3,4,2,2};
int number=duplicate(Arr);
cout<<"Duplicate number is :"<<number<<endl;
    
    
return 0;}
