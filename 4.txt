#include <iostream>
#include<vector>
using namespace std;

void merge(vector<int>&arr1, vector<int>& arr2, int m, int n){
    int total = m + n;
    int gap = (total + 1) / 2; 

    while (gap > 0) {
        int i = 0, j = gap;

        while (j < total) {
            int a, b;


          if (i < m) a = arr1[i];
            else a = arr2[i - m];

            if (j < m) b = arr1[j];
            else b = arr2[j - m];

            
      if (a > b) {
             if (i < m && j < m) 
      {swap(arr1[i], arr1[j]);}
                else if (i < m && j >= m) {swap(arr1[i], arr2[j - m]);}
                else {swap(arr2[i - m], arr2[j - m]);}
            }

            i++; j++;
        }

        if (gap == 1) gap = 0;
        else gap = (gap + 1) / 2;
    }
}

int main() {
    vector<int> arr1 = {1, 3, 5};
    vector<int> arr2 = {2, 4, 6};
    int m = arr1.size(), n = arr2.size();

    merge(arr1, arr2, m, n);

    cout << "arr1 = ";
    for (int x : arr1) cout << x << " ";
    cout << "\narr2 = ";
    for (int x : arr2) cout << x << " ";
    cout << endl;

    return 0;}
