#include <iostream>
#include<vector>
#include<string>
using namespace std;
string commonPrefix(vector<string>&Arr){
    int n=Arr.size();
    if(Arr.empty()){ 
       
        return "";}
    
    string prefix=Arr[0];
    for(int i=0;i<n;i++){
        while(Arr[i].find(prefix)!=0){
            prefix=prefix.substr(0,prefix.length()-1);
            if(prefix.empty()){
                return "";
            }
        }
    }
    return prefix;
}


int main() {
    vector<string>Arr={};
    string prefix=commonPrefix(Arr);
    cout<<"Common prefix is :"<<prefix<<endl;
    

    return 0;
}
